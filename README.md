# serve
Serve static files on http from command line

# Usage

    # serve current folder on port 8080
    serve 

    # serve on custom port and folder
    serve -port 1234 -folder /var/www
    
    # display help
    serve -h

# Install
    
    go get github.com/nilsmagnus/serve
